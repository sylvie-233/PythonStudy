### 快捷键
```yaml
快捷键:
    alt:
        ↑: # 移动代码行
        ↓: # 移动代码行
    ctrl:
        shift:
            [: # 折叠代码块
            ]: # 展开代码块
            .: # 当前内容切换
            p: # 命令面板
            enter: # 向上插入行
        enter: # 向下插入行
        `: # 打开终端
        d: # 重复选择
        f: # 查找
        g: # 跳转行
        k:
            ctrl:
                0: # 折叠所有代码块
                j: # 展开所有代码块
        l: # 选择行
```
