### 项目结构
```yaml
mkdocs:
    docs/:
    site/:
    mkdocs.yml:
```


#### mkdocs.yml
```yaml
mkdocs.yml:
    site_name:
    site_url:
    theme: # 主题
        name:
        icon:
            logo:
        palette:
            primary: # 主色调
        features:
            navigation.tabs:
            search.highlight:
    language:
        zh:
    nav: # 导航
    plugins: # 插件
    copyright:
    extra_javascript:
    extra_css:
```
