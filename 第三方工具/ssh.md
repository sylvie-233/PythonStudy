#### config
```yaml
~/.ssh/config:
    Host: # 主机别名
        HostName: # 真实服务器 IP/域名
        Username: # 默认用户
        Port: # SSH 端口
        IdentityFile: # 使用哪一个私钥
        ProxyJump: # 跳板机
        ServerAliveInterval: # 保持连接
        StrictHostKeyChecking: # 是否跳过指纹检查

```
